        public void Ajax() {
            set( "ajaxCheckUrl", to( AjaxCheck ) );
        }

        public void AjaxCheck() {
            String userName = ctx.Post( "userName" );
            if ("孙中山".Equals( userName )) {
                echoJsonMsg( "验证成功", true, null );
            }
            else {
                echoJsonMsg( "验证错误", false, null );
            }
        }
